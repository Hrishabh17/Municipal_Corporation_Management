create database if not exists dbms;

use dbms;

create table if not exists users(user_number bigint not null,user_name varchar(50),phone_number bigint,passcode varchar(16) not null,user_address varchar(150),primary key(user_number));

create table if not exists complaint(complaint_number bigint not null, user_id bigint not null, complaint_description varchar(150) ,complaint_photo varchar(150),complaint_status varchar(30) ,complaint_address varchar(150),completion_photo varchar(150),priority char(25),registered_time time,registration_date date,completion_time time,completion_date date,complaint_type varchar(25),complaint_subtype varchar(80),estimated_time bigint,primary key(complaint_number), foreign key(user_id)references users(user_number));

create table if not exists employee(ssn bigint not null,employee_name char(30) not null,designation char(30) ,employee_address varchar(150),ward char(5),sub_ward char(15),primary key(ssn));

create table if not exists registers(registered_Status char(20),complaint_number bigint not null,user_number bigint not null,foreign key(complaint_number) references complaint(complaint_number),foreign key(user_number) references users(user_number));

create table if not exists email(email_address varchar(40) not null,email_password varchar(75) not null,ssn bigint,user_number bigint,profile_photo varchar(70),foreign key(user_number) references users(user_number),foreign key(ssn) references employee(ssn));

create table if not exists department(department_number int not null,department_name char(20),ssn bigint not null,foreign key(ssn) references employee(ssn),primary key(department_name));

create table if not exists worker(worker_id bigint not null,worker_name char(30),worker_start_time datetime,worker_end_time datetime,no_of_tasks_assigned mediumint,primary key(worker_id));

create table if not exists comments(comment_description varchar(250),comment_time datetime,complaint_number bigint,ssn bigint,foreign key(complaint_number) references complaint(complaint_number),foreign key(ssn) references employee(ssn));
